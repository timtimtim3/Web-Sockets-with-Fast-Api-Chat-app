######
Big Picture: What this Ingress rule does

This config tells the Ingress controller (NGINX):

When a request comes to chatapp.local, route traffic to different services depending on the URL path.

So the traffic flow becomes:

Browser
   ↓
chatapp.local
   ↓
Ingress Controller (nginx)
   ↓
Service (backend)
   ↓
Backend Pods (3 replicas)

Ingress is basically the Kubernetes equivalent of your old NGINX reverse proxy.

# You can see the routing graphically:
kubectl get ingress -n chatapp
kubectl describe ingress chatapp-ingress -n chatapp

# We have:
/auth/*
/friends/*
/messages/*
/ws

# Local Manual Rolling Updates

# Backend changes
# After editing backend code:
docker build -t chat-backend:local -f ./server/Dockerfile .
kind load docker-image chat-backend:local
kubectl rollout restart deployment/backend -n chatapp
kubectl rollout status deployment/backend -n chatapp

# Frontend changes
# After editing frontend config/code:
docker build -t chat-frontend:local -f ./client/Dockerfile.k8s ./client
kind load docker-image chat-frontend:local
kubectl rollout restart deployment/frontend -n chatapp
kubectl rollout status deployment/frontend -n chatapp




# Horizontal Pod Autoscaling (HPA)

# install metrics server
kubectl apply -f https://github.com/kubernetes-sigs/metrics-server/releases/latest/download/components.yaml

# patch for local:
kubectl patch deployment metrics-server -n kube-system \
  --type='json' \
  -p='[{"op":"add","path":"/spec/template/spec/containers/0/args/-","value":"--kubelet-insecure-tls"}]'

# check:
kubectl get pods -n kube-system
kubectl top nodes
kubectl top pods -n chatapp

# Update backend yaml with resources, requests, limits, cpu, memory section…
# Reapply
kubectl apply -f k8s/local/backend-deployment.yaml
kubectl rollout status deployment/backend -n chatapp

# apply the HPA
kubectl apply -f k8s/local/backend-hpa.yaml
kubectl get hpa -n chatapp





# Generate load: Now we need traffic.
# A simple approach is to use a temporary load pod inside the cluster.
kubectl run -n chatapp load-generator --rm -it --image=busybox -- /bin/sh

# Then inside it:
while true; do wget -q -O- http://backend:8000/docs > /dev/null; done

# if exited, re-enter like this:
kubectl exec -it -n chatapp load-generator -- /bin/sh

# exit:
CTRL + C
exit
# then, delete:
kubectl delete pod load-generator -n chatapp
